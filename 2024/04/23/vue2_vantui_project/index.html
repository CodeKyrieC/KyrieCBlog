返回顶部组件封装

直接修改元素的位置没有过渡效果，不方便操作dom的css属性添加过渡属性，使用计时器实现滑动效果

<template>
    <div>
      <!--返回顶部-->
      <van-button v-if="btnFlag" icon="back-top" type="primary" @click="backTop" />
    </div>
  </template>
  
  <script>
  export default &#123;
    name: 'BackTop',
    data () &#123;
      return &#123;
        btnFlag: false // 默认按钮不显示
      &#125;
    &#125;,
    // vue的两个生命钩子，这里不多解释。
    // window对象，所有浏览器都支持window对象。它表示浏览器窗口，监听滚动事件
    mounted () &#123;
      window.addEventListener('scroll', this.scrollToTop)
    &#125;,
    // 页面离开，取消addEventListener，不然页面会报错
    destroyed () &#123;
      window.removeEventListener('scroll', this.scrollToTop)
    &#125;,
    methods: &#123;
      // 点击图片回到顶部方法，加计时器是为了过渡顺滑
      backTop () &#123;
        const that = this
        const timer = setInterval(() => &#123;
          const ispeed = Math.floor(-that.scrollTop / 5)
          document.documentElement.scrollTop = document.body.scrollTop = that.scrollTop + ispeed
          if (that.scrollTop === 0) &#123;
            clearInterval(timer)
          &#125;
        &#125;, 16)
      &#125;,
      // 为了计算距离顶部的高度，当高度大于60显示回顶部图标，小于60则隐藏
      scrollToTop () &#123;
        const that = this
        const scrollTop = window.pageYOffset || document.documentElement.scrollTop || document.body.scrollTop
        that.scrollTop = scrollTop
        if (that.scrollTop > 0) &#123;
          that.btnFlag = true
        &#125; else &#123;
          that.btnFlag = false
        &#125;
      &#125;
    &#125;
  &#125;
  </script>

      filterCategory: [], // 搜索结果分类
      goodsList: [], // 搜索结果列表

    &#125;

  created () &#123;

      &#125;).then(res => &#123;
        // console.log(res.data)

        // 直接解构赋值的写法
        // let &#123; filterCategory, goodsList &#125; = res.data

        const newArr = JSON.parse(JSON.stringify(res.data.filterCategory).replace(/name/g, 'text').replace(/id/g, 'value'))
